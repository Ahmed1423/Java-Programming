class Sum {
    
    int sum(int a, int b) {
        return a + b;
    }
    
    int sum(int a, int b, int c) {
        return a + b + c;
    }
}

class Main {
    
    public static void main(String[] args) {

        Sum sum = new Sum();
        
        System.out.println(sum.sum(2, 4));
        System.out.println(sum.sum(2, 4, 5));
    }
}
