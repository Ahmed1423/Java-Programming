import java.io.*;
import java.util.*;
import java.util.function.*;

class Main {
    
    static void varArgs(int ...args) {
        
        for(int i = 0; i < args.length; ++i) {
            System.out.print(args[i] + " ");
        }
        System.out.println();
    }
    public static void main(String[] args) {
        
        varArgs(1, 2);
        varArgs(1, 2, 3, 4, 5, 6);
    }
}
